	alsoletter={1234567890, :,\%},%
	otherkeywords={\%, \}, \{, \&, \|},
	alsoother={\#},
	literate=*%
		{:}{{\literatecolour:}}{1}%

		{*}{{\literatecolour*}}{1}%

      }, %DeviceToHost, HostToDevice,
